	type = character_event

	title = become_god_elden_ring.0001.title
	theme = war

	left_portrait = {
		character = ROOT

	}
	
	immediate = {
		remove_trait = empyrean_elden_ring

			modifier = religious_reformer_modifier
		}

			base = 10 
		}
	}
	
}
