give_grace_elden_ring_interaction = {
	category = interaction_category_friendly
	icon = alliance

	desc = give_grace_elden_ring_interaction_desc
	
	auto_accept = yes
	
	ai_targets = {
		ai_recipients = vassals
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_frequency = 12

	is_shown = {
		scope:actor = {
			has_trait = god_elden_ring
		}
		scope:recipient = {
			NOT = { this = scope:actor }
		}
	}
	
	on_accept = {
		scope:actor = {
			
		}
		scope:recipient = {
			add_character_modifier = {
				modifier = grace_given
				years = 10
			}
			add_opinion = {
				target = root
				modifier = friendliness_opinion
				opinion = 30
			}
		}
	}
	

	cost = {
		piety = 100
	}
}
