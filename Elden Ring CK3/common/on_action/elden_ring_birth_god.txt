on_action_ai_become_empyrean = {
	effect = {
		if = { # emperor or more that is VERY religious
			limit = {
				AND = {
					is_independent_ruler = yes
					NOR = {
						has_trait = empyrean_elden_ring
						#has_trait = demigod_elden_ring
						has_trait = god_elden_ring
					}
					highest_held_title_tier >= tier_empire
					piety_level >= 5
				}
			}
			random_list = {
				5 = {
					add_trait = empyrean_elden_ring
				}
				95 = {
				
				}		
			}
		}
				
	}
}

on_action_gods_or_dragons_immune_to_old_age = {
	effect = {
		if = {
			limit = {
				OR = {
					has_godhood_elden_ring = yes
					has_trait = ancient_dragon_elden_ring
					has_trait = lesser_dragon_elden_ring
				}
				
			}
			remove_trait = disfigured
			remove_trait = incapable
			remove_trait = cancer
		}
				
	}
}

on_action_special_tarnished_produce_normal_ones = {
	effect = {
		if = {
			limit = {
				OR = {
					scope:real_father = {
						has_trait = special_tarnished
					}
					scope:mother = {
						has_trait = special_tarnished
					}
				}
			}
			add_trait = tarnished_elden_ring			
		}
	}
}

on_action_marais_children_born_sick = {
	effect = {
		if = {
			limit = {
				is_male = yes
				dynasty = dynasty:dynn_marais
			}
			add_trait = sickly			
		}
	}
}
