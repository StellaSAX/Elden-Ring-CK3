	group = "crimes"
	doctrine_grafting_crime = {
		piety_cost = {
			value = faith_doctrine_cost_low
			if = {
				limit = { has_doctrine = doctrine_grafting_crime }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			else_if = {
				limit = { has_doctrine = doctrine_grafting_accepted }
				multiply = faith_changed_doctrine_cost_mult_two_step
